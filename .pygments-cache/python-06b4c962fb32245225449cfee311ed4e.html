def C(l,m):
    if type(l)==type([]):
        n=len(l)
    elif type(l)==type(0) : n=l;l=range(1,n+1)
    else : raise TypeError

    i=1;s=2;li=[1]*(m+1)
    L=[]

    while s!=0:
        if s!=m :
           if li[s] != n+s-m :
               li[s]+=1
               li[s+1]=li[s]
               s+=1
           else : s-=1

        else :
           if li[s] !=n :
               li[s]+=1

               L+=[[l[k-1] for k in li[1:]]]

           else : s-=1
    return L
